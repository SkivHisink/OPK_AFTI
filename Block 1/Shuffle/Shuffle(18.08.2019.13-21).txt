void randomize_array(int array[], int n)
{
	int* c;
	int i = 0;
	int temp;
	c = (int*)malloc(n * sizeof(int));
	while (i<n) {
		c[i] = rand() % 10;
		for (int j = 0; j < n; j++) {
			if (c[j] == c[i] && i != j) {
				i--;
				j = n + 1;
			}
		}
		i++;
	}
	for (int j = 0; j < n; j++) {
		temp = array[c[j]];
		array[c[j]] = array[j];
		array[j] = temp;
	}
}
