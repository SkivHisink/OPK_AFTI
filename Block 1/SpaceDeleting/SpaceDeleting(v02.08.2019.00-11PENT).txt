#include <stdio.h>
#include <malloc.h>
#include <ctype.h>

void strtrim(char* string);

int main()
{
	char a[6] = " abc ";
	strtrim(a);
	return 0;
}

void strtrim(char* string)
{
	int counter = 0;
	while (string[counter] != '\0') {
		counter++;
	}
	if (isspace(string[0])) {
		for (int i = 0; i < counter; i++) {
			string[i] = string[i + 1];
		}
	}
	if (isspace(string[counter - 2]))
	{
		string[counter - 2] = '\0';
	}
	printf("%s", string);
}
